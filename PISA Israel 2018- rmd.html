First, Let's create a distance function between two classes (class-level variables):: The class variables are: Categorical variable: System (distance Calculated as 0.25 for each step in the range [Israel Arabic- Israel Ultra Orthodox Girls]). Numeric variables: STRATIO, TOTAT, CLSIZE (Descrete), EDUSHORT and STAFFSHORT. I used L2 norm to calculate the difference between two observations for the numeric.

After filling in the missing values I divided each variable by it's standard deviation and calculated the L2 norm as the distance, As mentioned above.

class_distance = function(ob_a, dta){
  dis = 0
  
  #distance between systems- same system = 0, I made a range of systems where orthodox is one edge and arabic is the other, the distance between systems is between 0 and 1
  sys_dist = c("Israel Ultra Orthodox Girls", "Israel Hebrew Religious", "Israel Hebrew Secular", "Israel Arabic")
  
  #Make the systems numeric (as ordinal)
  dta = dta %>% mutate(System_ord = ifelse(System == "Israel Ultra Orthodox Girls", 1,
                      ifelse(System == "Israel Hebrew Religious",2,
                      ifelse(System == "Israel Hebrew Secular", 3,
                      ifelse(System == "Israel Arabic", 4, NA)))))
  
  dis = dis + abs(which(ob_a$System == sys_dist) - dta$System_ord)/4

  #numeric variables:
  dis_class_vars = c("STRATIO", "TOTAT", "CLSIZE", "EDUSHORT", "STAFFSHORT")
  #The sd of each column
  sd_class_vars = c(sd(pisa$STRATIO), sd(pisa$TOTAT), sd(pisa$CLSIZE), sd(pisa$EDUSHORT), sd(pisa$STAFFSHORT))
  
  #   #The mean of each column, not needed because it falls in the distance function when using oba-obb
  # mean_class_vars = c(mean(pisa$DISCLIMA, na.rm = T), mean(pisa$TEACHSUP, na.rm = T), mean(pisa$STRATIO, na.rm = T), mean(pisa$TOTAT, na.rm = T), mean(pisa$CLSIZE, na.rm = T), mean(pisa$EDUSHORT, na.rm = T), mean(pisa$STAFFSHORT, na.rm = T))
  # 
  
  
  #take the relevant variables and replace NAs with the mean of the variable- if both obs have the same missing value- than the distance of this variable is 0
  ob_a = ob_a[,dis_class_vars]
  dta = dta[,dis_class_vars]
  
  
  
  
  #The distance function is thesqrt of the sum of the distances while I chose to impute the NA values by averages by group
  dis = dis +  sqrt(apply((((t(ob_a)-dta)/sd_class_vars)^2), 1, sum))
  
  return(dis)
}

Now, Let's create a distance function between a student and a dataframe of students (student-level variables): The student variables are: Categorical variable: language (the distance is 1 if not the same language spoken at home) Ordinal variables: EFFORT1 and DURECEC (the distance is the L1 norm). Numeric variables: AGE, PAREDINT, ESCS, JOYREAD, DISCLIMA AND TEACHSUP. I used diveded by the standard deviation and used square root of the L2 norm to calculate the difference between two observations for the numeric variables.

The variable "PAREDINT" has some values of 3, 6,9 and many has 16 so the distance might be too big than what I want the weight for this variable be, therefore I will use the log of the distance between the observations (just for this variable).

student_distance = function(ob_a, dta){

  #just to make sure these are dataframes just like pisa (with names)

  ob_a = as.data.frame(ob_a)
  if(length(ob_a) != 18){
    ob_a$PVREAD = NA
  }

  
  dta = as.data.frame(dta)
  if(length(dta) != 18){
    dta$PVREAD = NA
  }

  names(ob_a) = names(pisa)
  names(dta) = names(pisa)
  
  #start the distance variable
  dis = 0
  
  #numeric variables:
  dis_student_vars_numeric = c("AGE", "PAREDINT", "ESCS", "JOYREAD", "DISCLIMA", "TEACHSUP")

  ## dis_student_categ = c("EFFORT1", "DURECEC", "language")
  #Add 1 if not same language
  dis = dis + as.numeric(ob_a$language!=dta$language)
  
  #Make the ordinal variables numeric 

  #Add L1 norm of ordinal variables (diveded by the standard deviation of the variable)
  dis = dis + abs(ob_a$EFFORT1- dta$EFFORT1)/sd(pisa$EFFORT1) + abs(ob_a$DURECEC- dta$DURECEC)/sd(pisa$DURECEC)
  
  #The sd of each numeric column
  sd_student_vars = c(sd(pisa$AGE), sd(pisa$PAREDINT), sd(pisa$ESCS), sd(pisa$JOYREAD), sd(pisa$DISCLIMA), sd(pisa$TEACHSUP))

  #take the relevant variables and replace NAs with the mean of the variable- if both obs have the same missing value- than the distance of this variable is 0
  ob_a = ob_a[,dis_student_vars_numeric]
  dta = dta[,dis_student_vars_numeric]

  
  #use sqrt of the L2 norm for the numeric variables, use the log of the distance for PAREDINT
  dd = (ob_a$PAREDINT-dta$PAREDINT)/sd(pisa$PAREDINT)

  dis = dis +  as.vector(sqrt(apply((((t(ob_a)-dta)/sd_student_vars)^2), 1, sum))) + 
    abs(dd) 
  
  return(dis)
}

Now I will create a function that calculated the distance between an observation and a dataframe with alpha as the weight of the student-level distance

calculate_distance = function(dta, ob_a, alph, student_distance, class_distance){
  return(alph*student_distance(ob_a, dta) + (1-alph)*class_distance(ob_a, dta))
}

Here I will create a function that for a given observation, K and alpha- will return the prediction using my KNN function.

predict_KNN = function(ob, dta, K, alph, student_distance, class_distance){

  
  if(nrow(dta) < K){
    return(mean(dta$PVREAD))
  }
  
  
  distances = calculate_distance(dta = dta, ob_a = ob, alph = alph, student_distance = student_distance, class_distance = class_distance)

  min_vals = sort(distances)[1:K]
  min_ind = which(distances %in% min_vals)
  
  prediction_KNN = mean(dta$PVREAD[min_ind])
  return(prediction_KNN)
}



predict_multi_KNN = function(test, train, K, alph, student_distance, class_distance){
  
  predictions = c()
  for(i in c(1:nrow(test))){
    predictions = append(predictions, predict_KNN(test[i,], train, alph = alph, K = K, student_distance = student_distance, class_distance = class_distance))
  }
  
  return(predictions)
}

RMSE = function(res_guess, test_pv){
  return(sqrt(mean((res_guess-test_pv)^2)))

Cross-Validation

In order to choose the best \(\alpha\) and K, I will use P-Fold-Cross-Validation (K fold but I already have variable K) with various Ks and \(\alpha\)s. for each, I will calculate the RMSE and visualize it with a plot to decide which combination is the best.

#P is the number of folds in the P-fold Cross Validation
uniform_CV = function(dta, P){
  
  errs = c()
  a_cv = c()
  k_cv = c()
  

  Ks = c(10:60)
  alphs = c(0.15, 0.3, 0.7, 0.9)
  P = 10 #10 Fold Cross Validation  (P = 10)
  for(a in alphs){
    for(k in Ks){
      
      ers = c()
      resample = dta[sample(c(1:nrow(dta)), size = nrow(dta)),]

      for(i in c(1:P)){
        rng = c(as.integer(floor(nrow(resample)/P)*(i-1)+1) : as.integer(floor(nrow(resample)/P*i)))
        # rng = floor(nrow(resample)*0.9) #If I want to use train/test with 1foldCV
        train = resample[-rng,]
        test = resample[rng,]
        res = predict_multi_KNN(test %>% select(-PVREAD), train = train, K = k, alph = a, student_distance= student_distance, class_distance = class_distance)
        ers = append(ers, RMSE(res, test$PVREAD))
      }
      errs = append(errs, mean(ers))
      #just to make it easier to make the plot later
      a_cv = append(a_cv, a)
      k_cv = append(k_cv, k)
    }

  }
  
  
  # res = predict_multi_KNN(test %>% select(-PVREAD), pisa[-sam,], K = K, alph = alph, student_distance= student_distance, class_distance = class_distance)
  r = as.data.frame(cbind(errs, a_cv, k_cv))
  names(r) = c("errs", "alphas", "Ks")
  return(r)

}
